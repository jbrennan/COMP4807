	public void recievedDataFromStation(int stationId, String data)
	{
		// Station 3 == Cares about blue blocks being dropped off by station 2
		// Station 1 == cares about red blocks
		
		System.out.printf("Station ID: %d, Data: [ %s ]\n", stationId, data);
		
		if(stationId != 2)
		{
			return;
		}
		
		StationMessage message = StationMessage.Parse(stationId, data);
		switch(message.GetMessageType())
		{
		case STATION_2_DONE:
			// Add Code to Tell Robot to go pickup blocks
			break;
		case STATION_2_BLUE_BLOCK_DROPPED_OFF:
			// Record the point
			int x = message.GetX();
			int y = message.GetY();
			System.out.printf("Station 2 Dropped Block off at [X,Y=(%d, %d)]", x, y);
			break;
		case STATION_2_RED_BLOCK_DROPPED_OFF:
			// Don't care
			break;
		default:
			// Don't care about anything else
			return;
		}
	}
